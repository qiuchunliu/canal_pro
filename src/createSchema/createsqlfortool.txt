CREATE TABLE `borrower` (

  `borrower_uuid` varchar(36) CHARACTER SET utf8 COLLATE utf8_unicode_ci DEFAULT NULL COMMENT '贷款人uuid',
  `borrower_name` varchar(100) CHARACTER SET utf8 COLLATE utf8_unicode_ci DEFAULT NULL COMMENT '姓名',
  `certificate_type` int(11) unsigned DEFAULT NULL COMMENT '证件类型',
  `certificate_no` varchar(255) CHARACTER SET utf8 COLLATE utf8_unicode_ci DEFAULT NULL COMMENT '证件号码',
  `gmt_create` datetime DEFAULT NULL COMMENT '创建时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `borrower_uuid` (`borrower_uuid`),
  KEY `idx_certificate_no` (`certificate_no`) USING BTREE,
  KEY `idx_gmt_create` (`gmt_create`) USING BTREE
) ENGINE=InnoDB AUTO_INCREMENT=165 DEFAULT CHARSET=utf8 COMMENT='贷款人信息表'
